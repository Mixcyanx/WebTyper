import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
if (s.hasNextInt()) {
int n = s.nextInt();
int max = 0;
for (int i = 0; i < n; i++) {
if (s.hasNextInt()) {
int val = s.nextInt();
if (val > max) max = val;
}
}
System.out.printf("%d\r\n", max);
}
}
}
