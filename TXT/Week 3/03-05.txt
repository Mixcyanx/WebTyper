import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
if (s.hasNextInt()) {
int x = s.nextInt();
int n = s.nextInt();
int lc = 0, rc = 0;
int ll = x, rl = x;
for (int i = 0; i < n; i++) {
int p = s.nextInt();
if (p < x) {
lc++;
if (lc == 1 || p < ll) ll = p;
} else if (p > x) {
rc++;
if (rc == 1 || p > rl) rl = p;
}
}
if (lc > rc) System.out.printf("%d %d\r\n", lc, ll);
else System.out.printf("%d %d\r\n", rc, rl);
}
}
}
