import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
if (s.hasNextInt()) {
int n = s.nextInt();
for (int i = 1; i <= 2 * n - 1; i++) {
int dist = Math.abs(n - i);
for (int j = 0; j < dist; j++) System.out.print(" ");
System.out.print("*");
if (dist < n - 1) {
int inner = 2 * (n - dist) - 3;
for (int j = 0; j < inner; j++) System.out.print(" ");
System.out.print("*");}
System.out.printf("\r\n");}}}}
