import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
while (s.hasNextInt()) {
int n = s.nextInt();
switch (n) {
case 7:
case 6:
System.out.printf("A+\r\n");
break;
case 5:
case 4:
System.out.printf("A\r\n");
break;
case 3:
System.out.printf("A-\r\n");
break;
case 2:
System.out.printf("B\r\n");
break;
case 1:
System.out.printf("C\r\n");
break;
case 0:
System.out.printf("F\r\n");
break;
}
}
}
}
