import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
while (s.hasNext()) {
String x = s.next();
int a = 0, b = 0;
for (int i = 0; i < x.length(); i++) {
int digit = x.charAt(x.length() - 1 - i) - '0';
if ((i + 1) % 2 != 0) {
a += digit;
} else {
b += digit;
}
}
System.out.printf("%d\r\n", Math.abs(a - b));
}
}
}
