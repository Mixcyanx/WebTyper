import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
long maxSoFar = -2000000000, currentMax = 0;
for (int i = 0; i < 10; i++) {
if (s.hasNextLong()) {
long x = s.nextLong();
currentMax += x;
if (currentMax > maxSoFar) maxSoFar = currentMax;
if (currentMax < 0) currentMax = 0;}}
System.out.print(maxSoFar + "\r\n");}}
