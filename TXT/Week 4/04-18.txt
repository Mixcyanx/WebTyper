import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
if (s.hasNextInt()) {
int n = s.nextInt();
String ns = String.valueOf(n);
String out = "";
int p = 1;
for (int i = ns.length() - 1; i >= 0; i--) {
int d = ns.charAt(i) - '0';
out += (out.isEmpty() ? "" : " ") + (d * p);
p *= 10;}
System.out.print(out + " ");}}}
