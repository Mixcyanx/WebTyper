import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
int c = 1;
while (s.hasNext()) {
int n = s.nextInt();
long[] a = new long[n];
for (int i = 0; i < n; i++) a[i] = s.nextLong();
long m = 0;
for (int i = 0; i < n; i++) {
long p = 1;
for (int j = i; j < n; j++) {
p *= a[j];
if (p > m) m = p;}}
System.out.println("Case #" + c + ": The maximum product is " + m + ".");
System.out.println();
c++;}}}
