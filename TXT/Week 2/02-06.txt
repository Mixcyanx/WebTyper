import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
if (s.hasNextInt()) {
int y = s.nextInt();
int m = s.nextInt();
int d = s.nextInt();
int[] days = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
if ((y % 4 == 0 && y % 100 != 0) || (y % 400 == 0)) days[1] = 29;
int sum = 0;
for (int i = 0; i < m - 1; i++) sum += days[i];
System.out.print(sum + d);
}
}
}
